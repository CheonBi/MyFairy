# 음성인식 API
- 아이들의 음성을 듣고 단어를 받아쓰려면 음성 -> text로 변환하는 과정이 필요하다.
- 따라서 음성인식 API인 whisper를 사용해보았다.

## 유튜브에서 회의 영상 음성 받아쓰기
- 일단 긴 버전의 음성을 사용하여 대략적인 시간과 받아쓰기 내용의 오차를 확인해보았다.
```python
from pydub import AudioSegment
import math

def cut_audio_in_chunks(audio_path, chunk_size,chunks_folder):
    track = AudioSegment.from_mp3(audio_path)
    chunk_len = chunk_size*60*1000
    chunks = math.ceil(len(track)/chunk_len)
    for i in range(chunks):
        start_time = i * chunk_len
        end_time = min((i+1) * chunk_len, len(track))
        track[start_time:end_time].export(f"{chunks_folder}/audio_chunk_{i}.mp3", format="mp3")

cut_audio_in_chunks("./files/audio.mp3", 10, "./files/chunks")

import openai
import glob

def transcribe_chunks(chunk_folder,destination):
  files = glob.glob(f"{chunk_folder}/*.mp3")
  final_script = ""
  for file in files:
    with open(file,"rb") as audio_file:
      response = openai.Audio.transcribe("whisper-1", audio_file)
      final_script += response["text"]
  with open(destination, "w") as file:
    file.write(final_script)

transcribe_chunks("./files/chunks", "./files/transcript.txt")
```
- pydub는 미디어 파일을 리스트처럼 활용할수 있도록 만들어진 라이브러리다.
